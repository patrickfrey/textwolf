00051 //   UCS-2 encoding is defined to be big-endian only. Although the similar designations 'UCS-2BE and UCS-2LE
00052 //   imitate the UTF-16 labels, they do not represent official encoding schemes. (http://en.wikipedia.org/wiki/UTF-16/UCS-2)
00053 //   therefore we take encoding=ByteOrder::BE as default.
00054 template <int encoding=ByteOrder::BE>
00055 struct UCS2

00057         enum
00058         {
00059                 LSB=(encoding==ByteOrder::BE),                  //< least significant byte index (0 or 1)
00060                 MSB=(encoding==ByteOrder::LE),                  //< most significant byte index (0 or 1)

00062                 Print2shift=(encoding==ByteOrder::LE)?8:0,      //< value to shift with to get the 2nd character to print
00063                 MaxChar=0xFFFF
00064         };
