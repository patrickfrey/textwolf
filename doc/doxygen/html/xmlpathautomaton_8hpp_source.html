104  void seekop( Operation op)
105  {
106  switch (op)
107  {
108  case OpenTag:
109  this->match( XMLScannerBase::OpenTag);
110  this->match( XMLScannerBase::HeaderStart);
111  break;
112  case CloseTag:
113  this->match( XMLScannerBase::CloseTag);
114  this->match( XMLScannerBase::CloseTagIm);
115  this->match( XMLScannerBase::HeaderEnd);
116  break;
117  case Attribute:
118  this->match( XMLScannerBase::TagAttribName);
119  this->match( XMLScannerBase::HeaderAttribName);
120  this->reject( XMLScannerBase::Content);
121  break;
122  case ThisAttributeValue:
123  this->match( XMLScannerBase::TagAttribValue);
124  this->match( XMLScannerBase::HeaderAttribValue);
125  this->reject( XMLScannerBase::TagAttribName);
126  this->reject( XMLScannerBase::HeaderAttribName);
127  this->reject( XMLScannerBase::Content);
128  this->reject( XMLScannerBase::OpenTag);
129  break;
130  case AttributeValue:
131  this->match( XMLScannerBase::TagAttribValue);
132  this->match( XMLScannerBase::HeaderAttribValue);
133  this->reject( XMLScannerBase::Content);
134  break;
135  case Content:
